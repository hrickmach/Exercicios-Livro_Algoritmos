PROGRAMA OPERACOES_MATEMATICAS
VAR
  A, B, RESULTADO : REAL
  OPCAO : INTEIRO
INICIO
  LEIA A, B, OPCAO
  CASO OPCAO
    SEJA 1 FAÇA 
      RESULTADO <- A + B
    SEJA 2 FAÇA
      RESULTADO <- A - B
    SEJA 3 FAÇA
      RESULTADO <- A * B
    SEJA 4 FAÇA
      RESULTADO <- A / B
  SENÃO
      ESCREVA "OPÇÃO INVÁLIDA"
  FIM_CASO
  ESCREVA "Resultado: ", RESULTADO
FIM
