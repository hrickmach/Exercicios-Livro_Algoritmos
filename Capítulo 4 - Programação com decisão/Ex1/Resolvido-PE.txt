PROGRAMA OPERACOES_MATEMATICAS
VAR
  A, B : REAL
  OPCAO : INTEIRO
INICIO
  LEIA A, B, OPCAO
  CASO OPCAO
    SEJA 1 FAÇA 
      ESCREVA A + B
    SEJA 2 FAÇA
      ESCREVA A - B
    SEJA 3 FAÇA
      ESCREVA <- A * B
    SEJA 4 FAÇA
      ESCREVA <- A / B
  SENÃO
      ESCREVA "OPÇÃO INVÁLIDA"
  FIM_CASO
FIM
