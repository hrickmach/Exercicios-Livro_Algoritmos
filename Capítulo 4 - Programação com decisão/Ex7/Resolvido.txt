PROGRAMA CONVERSAO_TEMPERATURA
VAR
  N, CONVERSAO : REAL
  OPCAO : INTEIRO
INICIO
  LEIA N
  ESCREVA "DESEJA CONVERTER PARA 1 - CELSIUS OU 2 - FAHRENHEIT?"
  LEIA OPCAO
  SE (OPCAO = 1) ENTÃO
    CONVERSAO <- (N - 32) * (5 / 9)
  SENÃO 
    CONVERSAO <- N * (9 / 5) + 32
  FIM_SE
  ESCREVA CONVERSAO
FIM
