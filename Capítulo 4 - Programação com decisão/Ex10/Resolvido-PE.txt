PROGRAMA MEDIA_PONDERADA

  NOTA1, NOTA2, NOTA3, MP: REAL

  LEIA NOTA1, NOTA2, NOTA3
  MP <- (NOTA1 * 2 + NOTA2 * 3 + NOTA3 * 5) / (2 + 3 + 5)
  SE (MP >= 7) ENTÃO
    ESCREVA "APROVADO"
  SENÃO
    SE (MP >= 5) ENTÃO
      ESCREVA "RECUPERAÇÃO"
    SENÃO
      ESCREVA "REPROVADO"
    FIM_SE
  FIM_SE
FIM
