PROGRAMA CONVERSAO_TEMPERATURA
VAR
  N, CONVERSAO : REAL
  MENU : CADEIA
  OPCAO : INTEIRO
INICIO
  LEIA N
  MENU <- "DESEJA CONVERTER PARA 1 - CELSIUS OU 2 - FAHRENHEIT?"
  ESCREVA MENU
  LEIA OPCAO
  SE (OPCAO = 1) ENTÃO
    CONVERSAO <- (N - 32) * (5 / 9)
    ESCREVA CONVERSAO
  SENÃO 
    SE (OPCAO = 2) ENTÃO
      CONVERSAO <- N * (9 / 5) + 32
    SENÃO
      ESCREVA "OPÇÃO INVÁLIDA"
      ESCREVA CONVERSAO
    FIM_SE
  SENÃO
    ESCREVA "OPÇÃO INVÁLIDA"
  FIM_SE
FIM
