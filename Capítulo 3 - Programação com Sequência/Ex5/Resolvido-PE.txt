PROGRAMA CUPOM_DESCONTO
VAR
  PRECO_PRODUTO, DESCONTO_PRODUTO : REAL
  CUPOM, DESCONTO_TOTAL, PRECO_FINAL : REAL
INICIO
  LEIA PRECO_PRODUTO, DESCONTO_PRODUTO, CUPOM
  DESCONTO_TOTAL <- (PRECO_PRODUTO * DESCONTO_PRODUTO / 100) + CUPOM
  PRECO_FINAL <- PRECO_PRODUTO - DESCONTO_TOTAL
  ESCREVA DESCONTO_TOTAL, PRECO_FINAL
FIM
